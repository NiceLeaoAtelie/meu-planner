<script>
    let tasks = [];

    // Carregar dados e iniciar atualiza√ß√£o visual
    window.onload = function() {
        const saved = localStorage.getItem('myDeadlineTasks');
        if (saved) {
            tasks = JSON.parse(saved);
            renderTasks();
        }
        // Atualiza a tela a cada 1 segundo para verificar atrasos em tempo real
        setInterval(renderTasks, 1000);
    };

    function addTask() {
        const name = document.getElementById('taskName').value;
        const deadline = document.getElementById('taskDeadline').value;

        if (!name || !deadline) {
            alert("Por favor, preencha o nome e a data/hora limite!");
            return;
        }

        const newTask = {
            id: Date.now(),
            name: name,
            deadline: deadline, // Salva como string ISO do input
            completed: false,
            createdAt: new Date().toISOString()
        };

        tasks.push(newTask);
        renderTasks();
        
        document.getElementById('taskName').value = '';
        document.getElementById('taskDeadline').value = '';
    }

    function renderTasks() {
        const list = document.getElementById('taskList');
        list.innerHTML = '';
        const now = new Date();

        tasks.forEach(task => {
            const deadlineDate = new Date(task.deadline);
            const isOverdue = now > deadlineDate && !task.completed;
            
            // Formatar data para exibi√ß√£o brasileira
            const formattedDeadline = deadlineDate.toLocaleString('pt-BR');

            const li = document.createElement('li');
            li.className = `task-item ${task.completed ? 'completed' : ''} ${isOverdue ? 'is-overdue' : ''}`;
            
            li.innerHTML = `
                <input type="checkbox" ${task.completed ? 'checked' : ''} onchange="toggleComplete(${task.id})">
                <div>
                    <span><strong>${task.name}</strong></span><br>
                    <span class="deadline-info">Prazo: ${formattedDeadline}</span>
                </div>
                <div class="timer-display ${isOverdue ? 'overdue-text' : ''}">
                    ${isOverdue ? 'üö® ATRASADA' : task.completed ? '‚úÖ CONCLU√çDA' : '‚è≥ EM DIA'}
                </div>
                <div class="controls">
                    <button onclick="deleteTask(${task.id})" style="background:#bdc3c7; color:white;">Remover</button>
                </div>
            `;
            list.appendChild(li);
        });
    }

    function toggleComplete(id) {
        const task = tasks.find(t => t.id === id);
        task.completed = !task.completed;
        renderTasks();
    }

    function deleteTask(id) {
        tasks = tasks.filter(t => t.id !== id);
        renderTasks();
    }

    function saveToStorage() {
        localStorage.setItem('myDeadlineTasks', JSON.stringify(tasks));
        alert("‚úÖ Prazos salvos com sucesso!");
    }

    function clearAll() {
        if (confirm("Deseja apagar todas as tarefas?")) {
            tasks = [];
            localStorage.removeItem('myDeadlineTasks');
            renderTasks();
        }
    }
</script>
